import CButton from '@/button';
import CCard from '@/card';

# 按钮 Button

按钮用来触发一些操作。

## 演示

<CCard class="space-x-2 space-y-2" title="按钮的基本使用">

按钮的 `type` 分别为 `default`、`primary`、`info`、`success`、`warning` 和 `error`。

  <CButton>Default</CButton>
  <CButton type="primary"> Primary </CButton>
  <CButton type="info">Info</CButton>
  <CButton type="success">Success</CButton>
  <CButton type="warning">Warning</CButton>
  <CButton type="error">Error</CButton>
</CCard>

<CCard class="mt-4 space-x-2 space-y-2" title="禁用">

给按钮设置 `disabled`

  <CButton disabled>Default</CButton>
  <CButton disabled primary>Primary</CButton>
  <CButton disabled info>Info</CButton>
  <CButton disabled success>Success</CButton>
  <CButton disabled warning>Warning</CButton>
  <CButton disabled error>Error</CButton>
</CCard>
